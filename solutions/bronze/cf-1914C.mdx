## Explanation

We can find the maximum experience point available by comparing the largest available experience point gained by completing one more quest or not.

## Implementation

**Time Complexity:** $\mathcal{O}(\min(n,k))$ for each test case

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
	int test_num;
	cin >> test_num;
	for (int t = 0; t < test_num; t++) {
		int n;
		int k;
		cin >> n >> k;

		vector<int> a(n);
		vector<int> b(n);
		for (int x = 0; x < n; x++) { cin >> a[x]; }
		for (int y = 0; y < n; y++) { cin >> b[y]; }

		int mx = b[0];
		// ans here represents the case only completing the first
		int ans = a[0] + mx * (k - 1);
		// anotherQuest is used to calculate the case completing other quests
		int another_Quest = a[0];
		// farthest location of available quest
		int limit = min(n, k);
		for (int z = 1; z < limit; z++) {
			mx = max(mx, b[z]);
			// compare experience point of completing one more quest or not
			if (ans < anotherQuest + a[z] + mx * (k - z - 1)) {
				ans = anotherQuest + a[z] + mx * (k - z - 1);
			}
			anotherQuest += a[z];
		}

		cout << ans << "\n";
	}
}
```
