```cpp
#include <bits/stdc++.h>
using namespace std;

// BeginCodeSnip{DSU (from the module)}
struct DSU {
	vector<int> p, sz;

	// stores history
	vector<pair<int &, int>> psnap, szsnap;

	DSU(int n) {
		p.resize(n);
		sz.resize(n, 1);
		iota(p.begin(), p.end(), 0);
	}

	int get(int x) { return (p[x] == x) ? x : get(p[x]); }

	void unite(int a, int b) {
		a = get(a);
		b = get(b);
		if (sz[a] < sz[b]) { swap(a, b); }

		// add to history
		szsnap.push_back({sz[a], sz[a]});
		psnap.push_back({p[b], p[b]});

		if (a != b) {
			p[b] = a;
			sz[a] += sz[b];
		}
	}

	void rollback() {
		szsnap.back().first = szsnap.back().second;
		szsnap.pop_back();
		psnap.back().first = psnap.back().second;
		psnap.pop_back();
	}
};
// EndCodeSnip

const int MAXN = 500000;
DSU dsu(MAXN);

void Init(int n) {}

void AddEdge(int u, int v) { dsu.unite(--u, --v); }

void RemoveLastEdge() { dsu.rollback(); }

int GetSize(int u) { return dsu.sz[dsu.get(--u)]; }
```
